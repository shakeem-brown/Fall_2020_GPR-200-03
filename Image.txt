// mainImage: process the current pixel (exactly one call per pixel)
//    fragColor: output final color for current pixel
//    fragCoord: input location of current pixel in image (in pixels)
void mainImage(out color4 fragColor, in sCoord fragCoord)
{
    // texture space [0, 1)
    // example: texture is 100 by 100 px, resInv = 0.01, 0.01
    // 1/100 of the whole texture !!!
    sDCoord pixelSizeInTextureSpace = 1.0 / iChannelResolution[0].xy;
    
    // setup
    // test UV for input image
    sCoord uv = fragCoord * pixelSizeInTextureSpace;
    
    sCoord uvUpRightCorner = (fragCoord + 0.5) * pixelSizeInTextureSpace;
    
    vec4 origImage = texture(iChannel0, uv);
    vec4 effect = texture(iChannel1, uv);
    
    vec4 compAdd = origImage + effect; //  
    vec4 compMix = vec4 (mix(origImage, effect, 1.0)); // the mix blend
    vec4 compScreen = 1.0 - (1.0 - origImage) * (1.0 - effect); // Screen blend formula
    vec4 compMult = origImage * effect; // Multiply blen formula
    
    // TESTING
    // set iChannel0 to 'Misc/Buffer A' and fetch sample
    fragColor = compScreen;
}
