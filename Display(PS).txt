layout (location = 6) out vec4 rtFragColor;

uniform sampler2D tex0;
uniform vec2 resolution;

in vec2 vTexCoord;

void main()
{
	vec2 uv = vTexCoord / resolution;
	vec4 image = texture(tex0, uv);
	rtFragColor = image;
}
