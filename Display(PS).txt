#ifdef GL_ES
precision highp float;
#endif // GL_ES

uniform sampler2D uTex;
uniform vec2 uResolution;

// VARYINGS
in vec2 vTexcoord;

layout (location = 0) out vec4 rtFragColor;

void main()
{
	vec2 uv = vTexcoord / uResolution;
	vec4 color = texture(uTex, uv);
	
	rtFragColor = color;
}
