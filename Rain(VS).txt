layout (location = 0) in vec4 aPos;
layout (location = 1) in vec3 aVelocity;
layout (location = 2) in vec4 aColor;
layout (location = 3) in float fLifeTime;
layout (location = 4) in float fSize;
layout (location = 5) in int jType;

// VARRYINGS PASSED TO GEOMETRY SHADER
out vec4 position;
out vec3 velocity;
out vec4 color;
out float lifetime;
out float size;
out int j;

void main()
{
	// PASSING TO GEOMETRY SHADER
	position = aPos;
	velocity = aVelocity;
	color = aColor;
	lifetime = fLifeTime;
	size = fSize;
	j = jType;
	gl_Position = aPos;;
}
