layout (location = 0) in vec4 aPos;
layout (location = 1) in vec3 aVelocity;
layout (location = 2) in vec4 aColor;
layout (location = 3) in float fLifeTime;
layout (location = 4) in float fSize;
layout (location = 5) in int jType;

// VARRYINGS PASSED TO GEOMETRY SHADER
out Vertex 
{
	vec4 position;
	vec3 velocity;
    vec4 color;
    float lifetime;
    float size;
    int j;
} vertex;

// TRANSFORM UNIFORMS
uniform mat4 projection; // uProjMat
uniform mat4 view; // uViewMat
uniform mat4 model; // uModelMat

// VARRYING

void main()
{
	// PASSING TO GEOMETRY SHADER
	vertex.position = aPos;
	vertex.velocity = aVelocity;
	vertex.color = aColor;
	vertex.lifetime = fLifeTime;
	vertex.size = fSize;
	vertex.j = jType;
	gl_Position = projection * view * model * aPos;
}
