//  OBJECT-SPACE
layout (location = 0) in vec4 aPositionS;

// TRANSFORM UNIFORMS
uniform mat4 uModelMat, uViewMat, uProjMat;

// VARYINGS
out vec4 vPosClip;

void main()
{
    // OBJECT POSITION PIPELINE
    mat4 modelViewMat = uViewMat * uModelMat;
	vec4 pos_view = modelViewMat * aPositionS;
	gl_Position = uProjMat * uViewMat * uModelMat * aPositionS;
	vPosClip = gl_Position; // view position clip is set equal to the position
}
