layout (location = 0) in vec4 aPositionD;
uniform mat4 uModelMat, uViewMat, uProjMat;

// VARYINGS
out vec2 vTexcoord;

void main()
{
	 // OBJECT POSITION PIPELINE
    mat4 modelViewMat = uViewMat * uModelMat;
	vec4 pos_view = modelViewMat * aPositionD;
	vec4 pos_clip = uProjMat * pos_view;
	gl_Position = uProjMat * uViewMat * uModelMat * aPositionD;
	
	vTexcoord = aPositionD.xy * 0.5 + 0.5;
	// NDC -> screen : [-1, +1] -> [0, 1]
}
