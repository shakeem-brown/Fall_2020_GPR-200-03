// Effect #0: Three Gradiant colors 
vec4 myEffect0(in vec2 fragCoord, in vec2 resolution)
{
    vec2 uv = fragCoord / resolution.xy; // Sets the position according to the resoultion
    vec3 color = vec3(uv, 0.0); // sets the color and gradian of the image

    float alpha = 1.0; // defines the alpha
    return vec4(color, alpha); // returns the gradiant image to main
}
