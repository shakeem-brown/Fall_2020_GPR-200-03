#ifdef GL_ES
precision highp float;
#endif // GL_ES

layout (location = 0) out vec4 rtFragColor;

uniform vec4 mouseClick;

void main()
{
	vec4 dropletColor = vec4(0.0, 1.0, 1.0, 1.0);
	
	if (mouseClick.z == 1.0)
	{
		dropletColor.r = 1.0;
	}
	
	rtFragColor = dropletColor;
}
